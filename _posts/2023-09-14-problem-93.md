    public int[] solution(String myString) {
        String[] mySplit = myString.split("x", myString.length());
        int[] answer = new int[mySplit.length];

        }
        return answer;
    }

import java.util.*;

class Solution {
    public String[] solution(String myString) {
        return Arrays.stream(myString.split("x"))
                     .filter(s -> !s.isEmpty())
                     .sorted()
                     .toArray(String[]::new);
    }
}
