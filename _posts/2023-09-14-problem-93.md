```java
class Solution {
    public int solution(String binomial) {
        int answer = 0;
        String[] calc = binomial.split(" ");
        
        int a = Integer.parseInt(calc[0]);
        String op = calc[1];
        int b = Integer.parseInt(calc[2]);
        
        answer = op.equals("+") ? a+b 
            : op.equals("-") ? a-b : a*b;
        
        return answer;
    }
}
```
