```java
import java.util.*;

class Solution {
    public int[] solution(int[] arr, int[] delete_list) {
        ArrayList<Integer> answer = new ArrayList<>();
        for (int list : delete_list) {
            answer.add(list);
        }

        return Arrays.stream(arr)
                .filter(list -> !answer.contains(list))
                .toArray();
    }
}
```
