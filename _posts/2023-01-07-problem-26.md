class Solution {
    public String solution(String my_string, int num1, int num2) {
        String answer = "";
        String[] index = my_string.split("");
        
        for(int i=0; i < index.length; i++) {
            if(i == num1) {
                answer += index[num2];
            } else if(i == num2) {
                answer += index[num1];
            } else answer += index[i];
        }
        return answer;
    }
}
