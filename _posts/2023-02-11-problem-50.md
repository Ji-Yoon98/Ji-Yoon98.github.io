class Solution {
    public String[] solution(String my_str, int n) {
        String[] answer = {};
        
        if(my_str.length() % n == 0) {
            answer = new String[my_str.length()/n];
        } else {
            answer = new String[my_str.length()/n+1];

        // substring ë¬¸ìì—´ ìë¥´ëŠ” ë©”ì„œë“œ (start, end)

            );
        }
        return answer;
    }
}
