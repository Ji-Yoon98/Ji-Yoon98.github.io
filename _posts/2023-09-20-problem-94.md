```java
import java.util.*;

class Solution {
    public String[] solution(String myStr) {
        String[] answer = Arrays.stream(myStr.split("[a,b,c]"))
            .filter(s -> !s.isEmpty())
            .toArray(String[]::new);
        
        return answer.length ==  0 ? new String[]{"EMPTY"} : answer;
    }
}
```

```java
import java.util.*;

class Solution {
    public List solution(int[] arr) {
        List<Integer> answer = new ArrayList<>();
        
        for (int num : arr) {
            for (int i = 0; i < num; i++) {
                answer.add(num);
            }
        }
        
        return answer;
    }
}
```
