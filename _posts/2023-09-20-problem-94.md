import java.util.*;

    public String[] solution(String myStr) {
        String[] answer = Arrays.stream(myStr.split("[a,b,c]"))
            .filter(s -> !s.isEmpty())
            .toArray(String[]::new);
        
        return answer.length ==  0 ? new String[]{"EMPTY"} : answer;
    }
}
