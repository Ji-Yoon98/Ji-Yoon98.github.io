class Solution {
    public String solution(String my_string) {
        String answer = "";
        String[] list = {"a","e","i","o","u"};
        
        for(int i=0; i < list.length; i++ ) {
            if(my_string.contains(list[i])) {
                answer = my_string.replaceAll("[aeiou]", "");
                my_string = answer;
            } else {
                answer = my_string;
            }
        }   
        return answer;
    }
}
