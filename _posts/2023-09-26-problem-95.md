```java
class Solution {
    public int solution(int[] arr1, int[] arr2) {
        int answer = 0;
        int sum1 = arr1.length, sum2 = arr2.length;
        
        if(sum1 == sum2) {
            for(int i=0; i < arr1.length; i++) {
                sum1 += arr1[i];
                sum2 += arr2[i];
                
                answer = sum1 > sum2 ? 1 : sum1 < sum2 ? -1 : 0;
            }
        } else {
            answer = sum1 > sum2 ? 1 : -1;
        }
        
        return answer;
    }
}
```
